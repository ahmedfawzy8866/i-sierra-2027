JavaScript: Scroll Timeline Animation
gsap.registerPlugin(ScrollTrigger);

const tl = gsap.timeline({
  scrollTrigger: {
    trigger: "#hero-trigger",
    start: "top top",
    end: "+=1200",
    scrub: 1.2,        // 1.2s catch-up smoothing
    pin: true,         // Lock section in viewport
    anticipatePin: 1
  }
});

// Chain animations sequentially
tl.to("#hero-title", { 
    opacity: 1, 
    y: 0, 
    duration: 2, 
    ease: "power4.out" 
  })
  .to("#hero-subtitle", { 
    opacity: 1, 
    y: 0, 
    duration: 1.5, 
    ease: "power3.out" 
  }, "-=1")
  .to("#hero-bg", { 
    clipPath: "inset(0% 0% 0% 0%)",
    scale: 1, 
    opacity: 0.9,
    duration: 4, 
    ease: "power2.inOut" 
  }, "-=1");

JavaScript: 3D Scene Setup & Interaction
const container = document.getElementById('webgl-canvas');
const scene = new THREE.Scene();
scene.background = new THREE.Color(0x0A1628);

// Camera + Renderer
const camera = new THREE.PerspectiveCamera(45, container.clientWidth / 400, 0.1, 1000);
camera.position.set(5, 5, 5);

const renderer = new THREE.WebGLRenderer({ antialias: true });
renderer.setSize(container.clientWidth, 400);
renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
container.appendChild(renderer.domElement);

// Orbit Controls (drag to rotate)
const controls = new THREE.OrbitControls(camera, renderer.domElement);
controls.enableDamping = true;
controls.dampingFactor = 0.05;
controls.maxPolarAngle = Math.PI / 2 - 0.05;

// Create wireframe apartment box
const geometry = new THREE.BoxGeometry(4, 2.5, 6);
const edges = new THREE.EdgesGeometry(geometry);
const lineMaterial = new THREE.LineBasicMaterial({ color: 0x2B5A9E });
const wireframe = new THREE.LineSegments(edges, lineMaterial);
scene.add(wireframe);

// Lighting
const light = new THREE.AmbientLight(0xffffff, 0.8);
scene.add(light);

// Animation loop
function animate() {
  requestAnimationFrame(animate);
  
  // Auto-rotate when idle (soft background rotation)
  if (controls.state === -1) {
    wireframe.rotation.y += 0.002;
  }
  
  controls.update();
  renderer.render(scene, camera);
}

animate();

// Responsive resizing
window.addEventListener('resize', () => {
  const width = container.clientWidth;
  camera.aspect = width / 400;
  camera.updateProjectionMatrix();
  renderer.setSize(width, 400);
});

Recommended Easing Rules
Entrance (fade-in, slide-in, scale-in):
  → ease: "power4.out"  // Fast start, slow finish

Exit (fade-out, slide-out):
  → ease: "power2.in"   // Slow start, fast finish

Continuous (bounce, shimmer, glow):
  → ease: "linear"      // Constant velocity

Interactive (hover, click):
  → ease: "power3.out"  // Natural, responsive feel
⚡ Performance Optimization Tips
✅ Do This:
• Use GPU-accelerated properties: transform, opacity only
• Set will-change: transform on animated elements
• Use requestAnimationFrame for smooth 60fps
• Limit simultaneous animations (max 5-8 per scroll section)
• Lazy-load Three.js canvas (only render when visible)

❌ Avoid This:
• Animating layout properties: width, height, left, right
• Creating 100+ DOM elements for parallax effects
• Re-rendering 3D scene on every scroll pixel
• Using setTimeout instead of requestAnimationFrame
• Animating box-shadow or blur without GPU acceleration
