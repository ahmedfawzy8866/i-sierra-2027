export async function syncVisualsToFirebase(
  sbrCode: string,
  visualFiles: File[],
): Promise<string[]> {
  const storage = getStorage();
  const urls: string[] = [];
 
  for (const file of visualFiles) {
    const storageRef = ref(storage, `assets/${sbrCode}/${file.name}`);
    const url = await getDownloadURL(storageRef);
    urls.push(url);
  }
 
  return urls;
}
